Update below code with your firebase configuraiton in index.js

`var firebaseConfig = {

    apiKey: "<API_KEY>",

    authDomain: "<PROJECT_ID>.firebaseapp.com",

    databaseURL: "https://<DATABASE_NAME>.firebaseio.com"

};`
